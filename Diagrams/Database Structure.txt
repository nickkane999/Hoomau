Users
    id: number (required)
    username: string [default to email value]
    bio: string
    phonenumber: number
    profession: string
    createdDate: date (required)

Script
    id: number (required)
    title: string (required)
    description: string (required)
    createduser: [user id reference] (required)
    code: string (required)
    createdDate: date (required)
    favorites: array
        userId (user id reference) (required)

ResultsData {
    id: string (required)
    createduser: [user id reference] (required)
    data: {
        [
            {
                field1: value1,
                field2: value2,
                ...
            },
            {
                field1: value1,
                field2: value2,
                ...
            }
        ]
    }
    createdDate: date (required)
}

Results
    title: string (required)
    filename: string (required)
    dataschema: {
        { 
            field1: type,
            field2: type,
            ...
        }
    } (required)
    dataresults: [Reference to ResultsData] (required)
    scriptFile: [Reference to Script] (required)
    userId: [Reference to User] (required)
    favorited: boolean (required)
    createdDate: date (required)
